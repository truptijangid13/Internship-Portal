<script>
    function loadStats(){
fetch("admin_stats.php")
.then(r => r.json())
.then(d => {
    document.getElementById("totalPrograms").innerText = d.programs;
    document.getElementById("totalMonthly").innerText = d.monthly;
    document.getElementById("totalWeekly").innerText = d.weekly;
    document.getElementById("totalReviews").innerText = d.reviews;

    new Chart(document.getElementById("adminChart"),{
        type:"bar",
        data:{
            labels:["Programs","Monthly","Weekly","Reviews"],
            datasets:[{
                data:[d.programs,d.monthly,d.weekly,d.reviews],
                backgroundColor:["#ff9900","#ffa41c","#ffcc80","#ffe0b2"]
            }]
        }
    });
});
}
    function navigate(page) {
    const contentArea = document.getElementById("content-area");

    switch(page) {

        // ================= MONTHLY TASK =================
        case 'addMonthlyTask':
            contentArea.innerHTML = `
                <h2>Monthly Task Entry</h2>

                <form method="POST" action="add_monthly_task.php">
                    <label>Title</label>
                    <input type="text" name="title" required>

                    <label>Description</label>
                    <input type="text" name="description" required>

                    <label>Month No</label>
                    <input type="number" name="month_no" required>

                    <button type="submit" class="form-btn">Save Monthly Task</button>
                </form>
            `;
        break;

        // ================= WEEKLY TASK =================
        case 'addWeeklyTask':
            contentArea.innerHTML = `
                <h2>Weekly Task Entry</h2>

                <form method="POST" action="add_weekly_task.php">
                    <label>Title</label>
                    <input type="text" name="title" required>

                    <label>Description</label>
                    <input type="text" name="description" required>

                    <label>Week No</label>
                    <input type="number" name="week_no" required>

                    <button type="submit" class="form-btn">Save Weekly Task</button>
                </form>
            `;
        break;

        // ================= VIEW PROGRAMS =================
        case 'viewPrograms':
            fetch("db_programs.php")
                .then(res => res.text())
                .then(data => contentArea.innerHTML = data)
                .catch(() => contentArea.innerHTML = "<p>Error loading programs.</p>");
        break;

        // ================= REVIEWS =================
        case 'review':
            fetch("db_reviews.php")
                .then(res => res.text())
                .then(data => contentArea.innerHTML = data)
                .catch(() => contentArea.innerHTML = "<p>Error loading reviews.</p>");
        break;

        // ================= ADD INTERNSHIP =================
        case 'contact':
            contentArea.innerHTML = `
                <h2>Add Internship</h2>

                <form method="POST" action="admin_page.php">
                    <label>Internship Title</label>
                    <input type="text" name="title" required>

                    <label>City</label>
                    <input type="text" name="city" required>

                    <label>Experience</label>
                    <input type="text" name="experience" required>

                    <label>Field</label>
                    <input type="text" name="field" required>

                    <label>Stipend</label>
                    <input type="text" name="stipend">

                    <label>Description</label>
                    <textarea name="description"></textarea>

                    <button type="submit" name="add_internship" class="form-btn">
                        Add Internship
                    </button>
                </form>
            `;
        break;

        // ================= DEFAULT =================
        default:
contentArea.innerHTML = `
<h2>Hello, LORD </h2>
<p>Select from navigation to begin.</p>

<div class="analytics-cards">
    <div class="card"><h4>Programs</h4><h2 id="totalPrograms">0</h2></div>
    <div class="card"><h4>Monthly Tasks</h4><h2 id="totalMonthly">0</h2></div>
    <div class="card"><h4>Weekly Tasks</h4><h2 id="totalWeekly">0</h2></div>
    <div class="card"><h4>Reviews</h4><h2 id="totalReviews">0</h2></div>
</div>

<div class="chart-box">
    <canvas id="adminChart"></canvas>
</div>
`;
loadStats();
break;
    }
}
